```go
// BE-08: Start CrossTrade dApp container (non-fatal).
// The compose file lives alongside the main docker-compose.yml in the
// deployment path.  We try to start it; failure only logs a warning so
// the L2 deployment is never blocked by the dApp.
crossTradeComposePath := filepath.Join(stack.DeploymentPath, "docker-compose.crosstrade.yml")
if _, statErr := os.Stat(crossTradeComposePath); statErr == nil {
    startOut, startErr := exec.CommandContext(ctx,
        "docker", "compose", "-f", crossTradeComposePath, "up", "-d",
    ).CombinedOutput()
    if startErr != nil {
        logger.Warn("failed to start CrossTrade dApp container (non-fatal)",
            zap.String("stackId", stackId.String()),
            zap.Error(startErr),
            zap.String("output", string(startOut)))
    } else {
        logger.Info("CrossTrade dApp container started",
            zap.String("stackId", stackId.String()),
            zap.String("composePath", crossTradeComposePath))
    }
} else {
    logger.Warn("docker-compose.crosstrade.yml not found, skipping dApp start (non-fatal)",
        zap.String("path", crossTradeComposePath))
}
```

Verify that `os` and `os/exec` are already imported in deployment.go. Both should already be present since `filepath.Join` and `exec.CommandContext` are used elsewhere. If `os` is missing, add it to the import block.

Non-fatal pattern: Matches existing patterns in deployment.go where .env.crosstrade write failure and metadata update failure are both logged as Warn but don't block deployment.
