   ```bash
   # Login

   # List stacks and find your stackId
   curl -s http://localhost:8000/api/v1/stacks/thanos \
     -H "Authorization: Bearer $TOKEN" | jq '.data.stacks[] | {id, chainName: .config.chainName}'

   # Check integrations (replace STACK_ID)
   curl -s http://localhost:8000/api/v1/stacks/thanos/STACK_ID/integrations \
     -H "Authorization: Bearer $TOKEN" | jq '.data.integrations[] | select(.type=="cross-trade")'
   # Expected: status "installed", metadata with contracts and tx hashes
   ```
