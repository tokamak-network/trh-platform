# AWS Credential Management Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Move AWS credential management from backend DB to Electron memory, supporting both CLI profile detection and SSO browser login.

**Architecture:** Electron main process reads `~/.aws/credentials` or performs OIDC device auth flow for SSO. Credentials stay in memory. On deployment, Electron injects them into the WebContentsView so trh-platform-ui sends them to backend in the deployment request. Backend uses them once and discards.

**Tech Stack:** `@aws-sdk/client-sso-oidc`, `@aws-sdk/client-sso`, Node.js `fs` for credential file parsing, Electron `shell.openExternal` for SSO browser flow.

```typescript
// src/main/aws-auth.ts
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import { shell } from 'electron';

export interface AwsProfile {
  name: string;
  source: 'credentials' | 'sso';
}

export interface AwsCredentials {
  accessKeyId: string;
  secretAccessKey: string;
  sessionToken?: string;
  source: string;
  expiresAt?: number; // epoch ms, for SSO tokens
}

interface SsoConfig {
  startUrl: string;
  region: string;
  accountId: string;
  roleName: string;
}

// In-memory credential store — never persisted
let activeCredentials: AwsCredentials | null = null;

function getCredentialsPath(): string {
  return process.env.AWS_SHARED_CREDENTIALS_FILE
    ?? path.join(os.homedir(), '.aws', 'credentials');
}

function getConfigPath(): string {
  return process.env.AWS_CONFIG_FILE
    ?? path.join(os.homedir(), '.aws', 'config');
}

function parseIniFile(filePath: string): Record<string, Record<string, string>> {
  if (!fs.existsSync(filePath)) return {};
  const content = fs.readFileSync(filePath, 'utf-8');
  const sections: Record<string, Record<string, string>> = {};
  let current = '';

  for (const line of content.split('\n')) {
    const trimmed = line.trim();
    if (!trimmed || trimmed.startsWith('#')) continue;
    const sectionMatch = trimmed.match(/^\[(.+)\]$/);
    if (sectionMatch) {
      current = sectionMatch[1];
      sections[current] = {};
      continue;
    }
    const kvMatch = trimmed.match(/^(\S+)\s*=\s*(.+)$/);
    if (kvMatch && current) {
      sections[current][kvMatch[1]] = kvMatch[2].trim();
    }
  }
  return sections;
}

export function listProfiles(): AwsProfile[] {
  const profiles: AwsProfile[] = [];

  // Standard credential profiles
  const creds = parseIniFile(getCredentialsPath());
  for (const name of Object.keys(creds)) {
    if (creds[name].aws_access_key_id) {
      profiles.push({ name, source: 'credentials' });
    }
  }

  // SSO profiles from config
  const config = parseIniFile(getConfigPath());
  for (const [key, values] of Object.entries(config)) {
    if (values.sso_start_url) {
      const name = key.startsWith('profile ') ? key.slice(8) : key;
      profiles.push({ name, source: 'sso' });
    }
  }

  return profiles;
}

export function loadProfile(name: string): AwsCredentials {
  const creds = parseIniFile(getCredentialsPath());
  const section = creds[name];
  if (!section?.aws_access_key_id) {
    throw new Error(`Profile "${name}" not found in credentials file`);
  }

  activeCredentials = {
    accessKeyId: section.aws_access_key_id,
    secretAccessKey: section.aws_secret_access_key,
    sessionToken: section.aws_session_token,
    source: `profile:${name}`,
  };
  return activeCredentials;
}

function getSsoConfig(profileName: string): SsoConfig {
  const config = parseIniFile(getConfigPath());
  const key = config[`profile ${profileName}`] ? `profile ${profileName}` : profileName;
  const section = config[key];
  if (!section?.sso_start_url) {
    throw new Error(`SSO profile "${profileName}" not found in config file`);
  }
  return {
    startUrl: section.sso_start_url,
    region: section.sso_region,
    accountId: section.sso_account_id,
    roleName: section.sso_role_name,
  };
}

export async function startSsoLogin(profileName: string): Promise<AwsCredentials> {
  const ssoConfig = getSsoConfig(profileName);

  // Lazy import to avoid loading AWS SDK when not needed
  const { SSOOIDCClient, RegisterClientCommand, StartDeviceAuthorizationCommand, CreateTokenCommand } =
    await import('@aws-sdk/client-sso-oidc');
  const { SSOClient, GetRoleCredentialsCommand } =
    await import('@aws-sdk/client-sso');

  const oidcClient = new SSOOIDCClient({ region: ssoConfig.region });

  // Step 1: Register client
  const registerRes = await oidcClient.send(new RegisterClientCommand({
    clientName: 'TRH-Desktop',
    clientType: 'public',
  }));

  // Step 2: Start device authorization
  const authRes = await oidcClient.send(new StartDeviceAuthorizationCommand({
    clientId: registerRes.clientId!,
    clientSecret: registerRes.clientSecret!,
    startUrl: ssoConfig.startUrl,
  }));

  // Step 3: Open browser for user to authenticate
  shell.openExternal(authRes.verificationUriComplete ?? authRes.verificationUri!);

  // Step 4: Poll for token (5s interval, max 5min)
  const maxAttempts = 60;
  const interval = authRes.interval ?? 5;
  let accessToken: string | undefined;

  for (let i = 0; i < maxAttempts; i++) {
    await new Promise(r => setTimeout(r, interval * 1000));
    try {
      const tokenRes = await oidcClient.send(new CreateTokenCommand({
        clientId: registerRes.clientId!,
        clientSecret: registerRes.clientSecret!,
        grantType: 'urn:ietf:params:oauth:grant-type:device_code',
        deviceCode: authRes.deviceCode!,
      }));
      accessToken = tokenRes.accessToken!;
      break;
    } catch (err: unknown) {
      const error = err as { name?: string };
      if (error.name === 'AuthorizationPendingException') continue;
      if (error.name === 'SlowDownException') {
        await new Promise(r => setTimeout(r, 5000));
        continue;
      }
      if (error.name === 'ExpiredTokenException') {
        throw new Error('SSO login timed out. Please try again.');
      }
      throw err;
    }
  }

  if (!accessToken) {
    throw new Error('SSO login timed out after 5 minutes.');
  }

  // Step 5: Get role credentials
  const ssoClient = new SSOClient({ region: ssoConfig.region });
  const roleRes = await ssoClient.send(new GetRoleCredentialsCommand({
    accessToken,
    accountId: ssoConfig.accountId,
    roleName: ssoConfig.roleName,
  }));

  const roleCreds = roleRes.roleCredentials!;
  activeCredentials = {
    accessKeyId: roleCreds.accessKeyId!,
    secretAccessKey: roleCreds.secretAccessKey!,
    sessionToken: roleCreds.sessionToken!,
    source: `sso:${profileName}`,
    expiresAt: roleCreds.expiration,
  };

  return activeCredentials;
}

export function getCredentials(): AwsCredentials | null {
  if (activeCredentials?.expiresAt && Date.now() > activeCredentials.expiresAt) {
    activeCredentials = null;
  }
  return activeCredentials;
}

export function clearCredentials(): void {
  activeCredentials = null;
}
```

function injectAwsCredentials(): void {
  if (!platformView) return;
  const creds = getAwsCredentials();
  if (!creds) return;

  const payload = {
    accessKeyId: creds.accessKeyId,
    secretAccessKey: creds.secretAccessKey,
    sessionToken: creds.sessionToken,
    source: creds.source,
  };

  platformView.webContents.executeJavaScript(
    `window.__TRH_AWS_CREDENTIALS__ = ${JSON.stringify(payload)};`
  ).catch(() => {});
}
